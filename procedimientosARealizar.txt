CREATE PROCEDURE BOBBY_TABLES.GET_CLIENTE_DNI
@DOC NUMERIC(10,0),
@TIPO INT
AS
BEGIN
	SELECT CLI_ID
	FROM BOBBY_TABLES.CLIENTES
	WHERE @TIPO = TIPO_DOC AND @DOC = NUMERO_DOC
	RETURN

END



CREATE PROCEDURE BOBBY_TABLES.GET_CLIENTE_MAIL
@MAIL VARCHAR(255)
AS
BEGIN
	SELECT CLI_ID
	FROM [BOBBY_TABLES].CLIENTES
	WHERE MAIL = @MAIL
	RETURN

END
